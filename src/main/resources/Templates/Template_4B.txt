# TEMPLATE 4B -- Qual foi a quantidade de ações do Itau negociadas no pregão de 05/05/2023?
SELECT ?O1 ?ANS WHERE {
    ?S1 P7 ?label .
    FILTER(REGEX(STR(?label), "#ENTIDADE_NOME#", "i"))
    ?S1 P1 ?SO1 .
    ?SO1 P2 ?codigo .
    ?codigo P8 ?O1 .
    ?SO1 P1* ?S3 .
    ?S3 P3 ?SO2 .
    ?SO2 P5 ?S2 .
    ?S2 P6 "#DATA#"^^xsd:date .
    ?SO2 #VALOR_DESEJADO# ?ANS .
}
